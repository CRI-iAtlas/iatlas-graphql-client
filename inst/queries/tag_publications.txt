query TagPublications(
  $dataSet: [String!]
  $related: [String!]
  $tag: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  tags(
    dataSet: $dataSet
    related: $related
    tag: $tag
    paging: $paging
    distinct: $distinct
  ){
    items {
      characteristics
      color
      longDisplay
      shortDisplay
      name
      publications {
        name
        journal
        firstAuthorLastName
        doId
        pubmedId
        title
        year
      }
    }
    paging {
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
