query Mutations(
  $entrez: [Int!]
  $mutationCode: [String!]
  $mutationId: [Int!]
  $mutationType: [String!]
  $sample: [String!]
  $status: [StatusEnum!]
) {
  mutations(
    entrez: $entrez
    mutationCode: $mutationCode
    mutationId: $mutationId
    mutationType: $mutationType
    sample: $sample
    status: $status
  ){
    id
    gene {
      entrez
      hgnc
    }
    mutationCode
    mutationType{
      name
      display
    }
  }
}
