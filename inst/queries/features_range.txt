query FeaturesRange(
  $dataSet: [String!]
  $related: [String!]
  $tag: [String!]
  $feature: [String!]
  $featureClass: [String!]
  $sample: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  features(
    dataSet: $dataSet
    related: $related
    tag: $tag
    feature: $feature
    featureClass: $featureClass
    sample: $sample
    paging: $paging
    distinct: $distinct
  ){
    items{
      name
      display
    	valueMax
    	valueMin
    }
    paging {
      type
      pages
      total
      startCursor
      endCursor
      hasPreviousPage
      hasNextPage
      page
      limit
      }
    error
  }
}
