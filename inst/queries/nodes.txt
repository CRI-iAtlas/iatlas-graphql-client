query Nodes(
  $dataSet: [String!]
  $entrez: [Int!]
  $feature: [String!]
  $maxScore: Float
  $minScore: Float
  $related: [String!]
  $network: [String!]
  $tag: [String!]
  $page: Int
) {
  nodes(
    dataSet: $dataSet
    entrez: $entrez
    feature: $feature
    maxScore: $maxScore
    minScore: $minScore
    related: $related
    network: $network
    tag: $tag
    page: $page
  ) {
    items{
      label
      name
      score
      x
      y
      tags {
        name
        shortDisplay
        longDisplay
      }
      gene{
        entrez
        hgnc
      }
      feature{
        name
        display
      }
    }
    page
    pages
    total
  }
}
