query PatientSlides(
  $barcode: [String!]
  $dataSet: [String!]
  $ethnicity: [EthnicityEnum!]
  $gender: [GenderEnum!]
  $maxAgeAtDiagnosis: Int
  $maxHeight: Float
  $maxWeight: Float
  $minAgeAtDiagnosis: Int
  $minHeight: Float
  $minWeight: Float
  $race: [RaceEnum!]
  $sample: [String!]
  $slide: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  patients(
    barcode: $barcode
    dataSet: $dataSet
    ethnicity: $ethnicity
    gender: $gender
    maxAgeAtDiagnosis: $maxAgeAtDiagnosis
    maxHeight: $maxHeight
    maxWeight: $maxWeight
    minAgeAtDiagnosis: $minAgeAtDiagnosis
    minHeight: $minHeight
    minWeight: $minWeight
    race: $race
    sample: $sample
    slide: $slide
    paging: $paging
    distinct: $distinct
  ){
    items{
      ageAtDiagnosis
      barcode
      ethnicity
      gender
      height
      race
      weight
      slides {
        name
        description
      }
    }
    paging {
      type
      pages
      total
      startCursor
      endCursor
      hasPreviousPage
      hasNextPage
      page
      limit
      }
    error
  }
}
