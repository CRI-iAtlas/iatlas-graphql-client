query TagSampleCount(
  $cohort: [String!]
  $dataSet: [String!]
  $related: [String!]
  $tag: [String!]
  $sample: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  tags(
    cohort: $cohort
    dataSet: $dataSet
    related: $related
    tag: $tag
    sample: $sample
    paging: $paging
    distinct: $distinct
  ){
    items{
      characteristics
      longDisplay
      shortDisplay
      color
      name
      sampleCount
    }
    paging{
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
