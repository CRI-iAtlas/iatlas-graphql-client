query CohortSelector(
  $dataSet: [String!]!
  $related: [String!]!
  $tag: [String!]
  $feature: [String!]
  $featureClass: [String!]
  $sample: [String!]
) {
  tags(
    dataSet: $dataSet
    related: $related
    tag: $tag
    feature: $feature
    featureClass: $featureClass
    sample: $sample
  ){
    characteristics
    longDisplay
    shortDisplay
    color
    name
    sampleCount
    samples
  }
}
