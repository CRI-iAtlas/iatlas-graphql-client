query TagRelated(
  $cohort: [String!]
  $dataSet: [String!]
  $related: [String!]
  $sample: [String!]
  $tag: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  tags(
    cohort: $cohort
    dataSet: $dataSet
    related: $related
    sample: $sample
    tag: $tag
    paging: $paging
    distinct: $distinct
  ){
    items {
      characteristics
      longDisplay
      shortDisplay
      color
      name
      related{
        characteristics
        color
        longDisplay
        shortDisplay
        name
      }
    }
    paging{
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
