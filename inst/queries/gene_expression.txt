query GeneExpression(
  $dataSet: [String!]
  $entrez: [Int!]
  $geneType: [String!]
  $maxRnaSeqExpr: Float
  $minRnaSeqExpr: Float
  $related: [String!]
  $sample: [String!]
  $tag: [String!]
) {
  genes(
    dataSet: $dataSet
    entrez: $entrez
    geneType: $geneType
    maxRnaSeqExpr: $maxRnaSeqExpr
    minRnaSeqExpr: $minRnaSeqExpr
    related: $related
    sample: $sample
    tag: $tag
  ){
    entrez
    hgnc
    samples{
      name
      rnaSeqExpr
    }
  }
}
