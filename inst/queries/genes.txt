query Genes(
    $dataSet: [String!]
    $entrez: [Int!]
    $geneType: [String!]
    $maxRnaSeqExpr: Float
    $minRnaSeqExpr: Float
    $related: [String!]
    $sample: [String!]
    $tag: [String!]
    $paging: PagingInput
    $distinct: Boolean
) {
  genes(
    dataSet: $dataSet
    entrez: $entrez
    geneType: $geneType
    maxRnaSeqExpr: $maxRnaSeqExpr
    minRnaSeqExpr: $minRnaSeqExpr
    related: $related
    sample: $sample
    tag: $tag
    paging: $paging
    distinct: $distinct
  ){
    items{
      entrez
      hgnc
      description
      friendlyName
      ioLandscapeName
      geneFamily
      geneFunction
      immuneCheckpoint
      pathway
      superCategory
      therapyType
    }
    paging {
      type
      pages
      total
      startCursor
      endCursor
      hasPreviousPage
      hasNextPage
      page
      limit
      }
    error
  }
}
