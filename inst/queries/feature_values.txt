query FeatureValues(
  $dataSet: [String!]
  $related: [String!]
  $tag: [String!]
  $feature: [String!]
  $featureClass: [String!]
  $sample: [String!]
  $maxValue: Float
  $minValue: Float
  $paging: PagingInput
  $distinct: Boolean
) {
  features(
    dataSet: $dataSet
    related: $related
    tag: $tag
    feature: $feature
    featureClass: $featureClass
    sample: $sample
    maxValue: $maxValue
    minValue: $minValue
    paging: $paging
    distinct: $distinct
  ){
    items{
      name
      display
      order
      class
      samples {
        name
        value
      }
    }
    paging {
      type
      pages
      total
      startCursor
      endCursor
      hasPreviousPage
      hasNextPage
      page
      limit
      }
    error
  }
}
