query FeatureValues(
  $cohort: [String!]
  $feature: [String!]
  $featureClass: [String!]
  $sample: [String!]
  $maxValue: Float
  $minValue: Float
  $paging: PagingInput
  $distinct: Boolean
) {
  features(
    cohort: $cohort
    sample: $sample
    feature: $feature
    featureClass: $featureClass
    maxValue: $maxValue
    minValue: $minValue
    paging: $paging
    distinct: $distinct
  ){
    items{
      name
      display
      order
      class
      samples {
        name
        value
      }
    }
    paging {
      type
      pages
      total
      startCursor
      endCursor
      hasPreviousPage
      hasNextPage
      page
      limit
      }
    error
  }
}
