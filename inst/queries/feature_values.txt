query FeatureValues(
  $cohort: [String!]
  $feature: [String!]
  $featureClass: [String!]
  $sample: [String!]
  $maxValue: Float
  $minValue: Float
  $paging: PagingInput
  $distinct: Boolean
) {
  featureValues(
    cohort: $cohort
    feature: $feature
    featureClass: $featureClass
    sample: $sample
    maxValue: $maxValue
    minValue: $minValue
    paging: $paging
    distinct: $distinct
  ){
    items{
      value
      sample {
        name
      }
      feature {
        name
        display
        class
        order
      }
    }
    paging {
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
