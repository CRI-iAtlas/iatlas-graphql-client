query GeneNodes(
  $dataSet: [String!]
  $entrez: [Int!]
  $maxScore: Float
  $minScore: Float
  $related: [String!]
  $network: [String!]
  $tag: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  nodes(
    dataSet: $dataSet
    entrez: $entrez
    maxScore: $maxScore
    minScore: $minScore
    related: $related
    network: $network
    tag: $tag
    paging: $paging
    distinct: $distinct
  ) {
    items{
      label
      name
      score
      x
      y
      tags {
        characteristics
        color
        name
        longDisplay
        order
        shortDisplay
      }
      gene {
        entrez
        hgnc
        friendlyName
      }
    }
    paging {
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
